    def port
      if port = host_with_port.split(/:/)[1]
        port.to_i
      elsif port = @env['HTTP_X_FORWARDED_PORT']
        port.to_i
      elsif @env.has_key?("HTTP_X_FORWARDED_HOST")
        DEFAULT_PORTS[scheme]
      else
        @env["SERVER_PORT"].to_i
      end
    end
