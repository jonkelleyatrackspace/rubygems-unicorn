    def call(env)
      path = env["PATH_INFO"]

      if can_serve(path)
        env["PATH_INFO"] = (path =~ /\/$/ ? path + @index : @urls[path]) if overwrite_file_path(path)
        @path = env["PATH_INFO"]
        apply_header_rules
        @file_server.call(env)
      else
        @app.call(env)
      end
    end
