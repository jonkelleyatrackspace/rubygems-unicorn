  def ssl(&block)
    ssl_require!
    before = @set[:listeners].dup
    opts = @set[:ssl_opts] = {}
    yield
    (@set[:listeners] - before).each do |address|
      (@set[:listener_opts][address] ||= {})[:ssl_opts] = opts
    end
    ensure
      @set.delete(:ssl_opts)
  end
